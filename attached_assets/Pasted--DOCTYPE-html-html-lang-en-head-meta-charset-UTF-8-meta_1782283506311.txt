function getUsers(){
  const s=localStorage.getItem('iss_users');
  let merged;
  if(!s){merged=[...DEFAULT_USERS];}
  else{
    const parsed=JSON.parse(s);
    merged=parsed.map(u=>{
      const def=DEFAULT_USERS.find(d=>d.username===u.username);
      // Refresh built-in seed fields (name/role/password) if code was updated,
      // but always keep the user's own edits: managedBy, department, email.
      if(def)return{...def,
        managedBy:u.managedBy!==undefined?u.managedBy:def.managedBy,
        department:u.department!==undefined?u.department:def.department,
        email:u.email!==undefined?u.email:def.email
      };
      return u;
    });
    for(const def of DEFAULT_USERS){if(!merged.find(u=>u.username===def.username))merged.push(def);}
  }
  // Safety net: anyone without a manager (and who isn't Boss) reports to Boss by default.
  merged=merged.map(u=>{
    if(u.username==='arvind.khot')return{...u,managedBy:null};
    if(!u.managedBy)return{...u,managedBy:'arvind.khot'};
    return u;
  });
  localStorage.setItem('iss_users',JSON.stringify(merged));
  return merged;
}
function saveUsers(u){localStorage.setItem('iss_users',JSON.stringify(u));}
function getTasks(){
  const s=localStorage.getItem('iss_tasks');
  if(!s){localStorage.setItem('iss_tasks',JSON.stringify(SEED_TASKS));return SEED_TASKS;}
  const existing=JSON.parse(s);
  const ids=new Set(existing.map(t=>t.id));
  const newSeeds=SEED_TASKS.filter(t=>!ids.has(t.id));
  if(newSeeds.length){const merged=[...existing,...newSeeds];localStorage.setItem('iss_tasks',JSON.stringify(merged));return merged;}
  return existing;
}
function saveTasks(t){localStorage.setItem('iss_tasks',JSON.stringify(t));tasks=t;}

function canViewAll(){return CU?.username==='arvind.khot';}

// Direct reports of a given username (one level down)
function getDirectReports(username){
  return users.filter(u=>u.managedBy===username);
}

// Everyone under a username in the hierarchy, at any depth (does NOT include the username itself)
function getAllReports(username){
  const result=[];
  const queue=getDirectReports(username);
  const seen=new Set();
  while(queue.length){
    const u=queue.shift();
    if(seen.has(u.username))continue;
    seen.add(u.username);
    result.push(u);
    queue.push(...getDirectReports(u.username));
  }
  return result;
}

// Self + everyone under them in the hierarchy
function getVisibleUsernames(username){
  return [username,...getAllReports(username).map(u=>u.username)];
}

function canViewTeam(){
  if(!CU)return false;
  return getDirectReports(CU.username).length>0; // anyone with at least 1 report has a Team view
}
function canAssign(){return !!CU;} // sabko assign

function visibleTasks(){
  if(!CU)return[];
  const mine=getVisibleUsernames(CU.username);
  return tasks.filter(t=>mine.includes(t.username));
}

function doLogin(){
  users=getUsers(); // always reload fresh
  const u=document.getElementById('inp-user').value.trim();
  const p=document.getElementById('inp-pass').value;
  const found=users.find(x=>x.username===u&&x.password===p);
  if(found){
    CU=found;
    localStorage.setItem('iss_cu',JSON.stringify(CU));
    document.getElementById('login-page').classList.add('hidden');
    document.getElementById('app').classList.remove('hidden');
    initApp();
  }else{
    const el=document.getElementById('login-err');
    el.textContent='Invalid username or password';
    el.classList.remove('hidden');
  }
}

function navigate(page){
  currentPage=page;
  document.querySelectorAll('.sb-link').forEach(el=>el.classList.remove('active'));
  const el=document.getElementById(`nav-${page}`);
  if(el)el.classList.add('active');
  tasks=getTasks();
  notifications=getNotifs();
  updateNotifBadge();
  const titles={dashboard:'Dashboard',tasks:'My Tasks',assign:'Assign Task',team:'Team View',credentials:'Credentials',reports:'Reports & Download',settings:'Settings'};
  document.getElementById('topbar-title').textContent=titles[page]||'';
  // destroy old charts
  Object.values(chartInstances).forEach(c=>{try{c.destroy();}catch(e){}});
  chartInstances={};
  const content=document.getElementById('content');
  if(page==='dashboard')renderDashboard(content);
  else if(page==='tasks')renderTasks(content);
  else if(page==='assign')renderAssign(content);
  else if(page==='team')renderTeam(content);
  else if(page==='credentials')renderCredentials(content);
  else if(page==='reports')renderReports(content);
  else if(page==='settings')renderSettings(content);
}

    const t=tasks.find(x=>x.id===id);

    const task=tasks.find(t=>t.id===id);

      <button class="btn btn-primary" onclick="submitTask('${taskId||''}')">Save</button>
    `);
  }
  window.submitTask=function(editId){
    const title=document.getElementById('tf-title').value.trim();
    if(!title){toast('Title required','err');return;}
    const data={
      title,description:document.getElementById('tf-desc').value,
      priority:document.getElementById('tf-priority').value,
      category:document.getElementById('tf-category').value,
      taskType:document.getElementById('tf-type').value,
      status:document.getElementById('tf-status').value,
      dueDate:document.getElementById('tf-date').value,
      dueTime:document.getElementById('tf-time').value,
      reminder:document.getElementById('tf-reminder').checked,

      }
    }else{

      const username=(canViewAll()&&unEl)?unEl.value||CU.username:CU.username;
      const nt={...data,id:uid(),username,createdAt:Date.now()};
      saveTasks([...tasks,nt]);
      toast('Task added','ok');
    }
    closeModal();
    renderTasks(el);
  }

  render();
}

function renderReports(el){
  const vt=visibleTasks();
  const now=new Date();
  const curMonth=now.toISOString().slice(0,7);

  // Generate last 12 months options
  const months=[];
  for(let i=0;i<12;i++){
    const d=new Date(now.getFullYear(),now.getMonth()-i,1);
    const val=d.toISOString().slice(0,7);
    const label=d.toLocaleString('default',{month:'long',year:'numeric'});
    months.push({val,label});
  }

  let rFilter={type:'month',month:curMonth,from:'',to:'',dept:'All',member:'All',category:'All'};

  function getFiltered(){
    let t=vt;
    // Date filter
    if(rFilter.type==='month'&&rFilter.month){
      t=t.filter(x=>x.dueDate&&x.dueDate.startsWith(rFilter.month));
    } else if(rFilter.type==='range'&&rFilter.from&&rFilter.to){
      t=t.filter(x=>x.dueDate>=rFilter.from&&x.dueDate<=rFilter.to);
    }
    // Dept filter
    if(rFilter.dept!=='All') t=t.filter(x=>getUserDept(x.username)===rFilter.dept);
    if(rFilter.member!=='All') t=t.filter(x=>x.username===rFilter.member);
    if(rFilter.category!=='All') t=t.filter(x=>x.category===rFilter.category);
    return t;
  }

  function downloadCSV(data){
    const headers=['Task Title','Description','Assigned To','Department','Priority','Category','Type','Status','Due Date','Due Time','Assigned By'];
    const rows=data.map(t=>[
      '"'+t.title.replace(/"/g,'""')+'"',
      '"'+(t.description||'').replace(/"/g,'""')+'"',
      '"'+getUserName(t.username)+'"',
      '"'+getUserDept(t.username)+'"',
      t.priority,t.category,t.taskType,t.status,t.dueDate,t.dueTime,

  function managerOptionsFor(u){
    // Can't manage self, and can't pick anyone who is already a report of u (would create a loop)
    const blocked=new Set([u.username,...getAllReports(u.username).map(x=>x.username)]);
    return users.filter(x=>!blocked.has(x.username));
  }

  function render(){
    el.innerHTML=`
      <div class="flex items-center justify-between" style="margin-bottom:4px">
        <div>

        <button class="btn btn-primary" onclick="openAddUser()">➕ Add New User</button>
      </div>
      <div class="card" style="overflow-x:auto;margin-top:16px">
        <table class="data-table">
          <thead><tr><th>Name</th><th>Role</th><th>Username</th><th>Password</th><th>Department</th><th>Reports To (Manager)</th><th>📧 Email (for auto reminders)</th></tr></thead>
          <tbody>
            ${users.map(u=>{
              const col=roleColors[u.role]||'#6b7280';
              const isBoss=u.username==='arvind.khot';

                <td style="font-weight:600">${u.name}</td>

            }).join('')}
          </tbody>
        </table>
      </div>
      <p style="color:var(--text3);font-size:11px;margin-top:10px">💡 Manager badalne se woh user us nayi manager ke "Team" aur "Reports" me dikhega, aur turant unka data download/view kar payega.</p>

  window.setEmail=function(username,newEmail){

    if(newEmail&&!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(newEmail)){

      render();
      return;

    const newUsers=users.map(u=>u.username===username?{...u,email:newEmail}:u);

      <button class="btn btn-primary" onclick="submitAddUser()">Add User</button>

  }

  window.submitAddUser=function(){
    const name=document.getElementById('au-name').value.trim();
    let username=document.getElementById('au-username').value.trim().toLowerCase();
    const password=document.getElementById('au-password').value;
    const role=document.getElementById('au-role').value.trim();
    const department=document.getElementById('au-dept').value;
    const managedBy=document.getElementById('au-manager').value;
    const email=document.getElementById('au-email').value.trim();
    const err=document.getElementById('au-err');
    err.classList.add('hidden');

    if(!name||!username||!password||!role){err.textContent='Sab required fields bharo (Name, Username, Password, Role)';err.classList.remove('hidden');return;}
    if(password.length<6){err.textContent='Password kam se kam 6 characters ka ho';err.classList.remove('hidden');return;}
    if(email&&!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)){err.textContent='Email format galat hai';err.classList.remove('hidden');return;}

    if(users.find(u=>u.username===username)){err.textContent='Yeh username already taken hai, dusra try karo';err.classList.remove('hidden');return;}

    const newUser={name,username,password,role,department,managedBy,email};
    const newUsers=[...users,newUser];
    saveUsers(newUsers);
    users=newUsers;
    closeModal();
    toast(name+' added successfully ✅','ok');
    render();
  }

  render();

function initEmailJS(cfg){
  if(typeof emailjs==='undefined')return false;
  if(!cfg.publicKey)return false;
  try{emailjs.init(cfg.publicKey);return true;}catch(e){return false;}
}

function buildTaskListText(taskList){
  if(taskList.length===0)return'No tasks scheduled.';
  return taskList.map((t,i)=>`${i+1}. ${t.title} [${t.priority}] — Due ${t.dueDate} ${t.dueTime}`).join('\n');
}

function sendTaskEmail(cfg,user,periodLabel,taskList){
  const params={
    to_email:user.email,
    to_name:user.name,
    period:periodLabel,
    task_count:String(taskList.length),
    task_list:buildTaskListText(taskList),
    company:'ISS Task Pro'
  };
  return emailjs.send(cfg.serviceId,cfg.templateId,params);
}

function runEmailScheduler(){
  const cfg=getEmailConfig();
  if(!cfg.enabled||!cfg.serviceId||!cfg.templateId||!cfg.publicKey)return;
  if(!initEmailJS(cfg))return;

  // Always check against the latest saved data, not just what's in memory for the current viewer.
  const freshTasks=getTasks();
  const freshUsers=getUsers();

  const now=new Date();
  const hh=String(now.getHours()).padStart(2,'0');
  const mm=String(now.getMinutes()).padStart(2,'0');
  const nowHM=hh+':'+mm;
  const todayStr=now.toISOString().split('T')[0];
  const weekKey=getWeekKey(now);
  const monthKey=now.toISOString().slice(0,7);
  const dow=String(now.getDay()); // 0=Sun..6=Sat

  const log=getEmailLog();
  let logChanged=false;
  const allUsers=freshUsers.filter(u=>u.email&&u.email.trim());

  // DAILY
  if(cfg.daily&&nowHM===cfg.dailyTime){
    allUsers.forEach(u=>{
      const key='daily_'+u.username+'_'+todayStr;
      if(log[key])return;
      const myDaily=freshTasks.filter(t=>t.username===u.username&&t.taskType==='daily'&&t.dueDate===todayStr);
      if(myDaily.length===0)return;
      sendTaskEmail(cfg,u,"Today's Daily Tasks",myDaily).then(()=>{},()=>{});
      log[key]=true;logChanged=true;
    });
  }
  // WEEKLY
  if(cfg.weekly&&dow===cfg.weeklyDay&&nowHM===cfg.weeklyTime){
    allUsers.forEach(u=>{
      const key='weekly_'+u.username+'_'+weekKey;
      if(log[key])return;
      const myWeekly=freshTasks.filter(t=>t.username===u.username&&t.taskType==='weekly'&&getWeekKey(new Date(t.dueDate))===weekKey);
      if(myWeekly.length===0)return;
      sendTaskEmail(cfg,u,"This Week's Weekly Tasks",myWeekly).then(()=>{},()=>{});
      log[key]=true;logChanged=true;
    });
  }
  // MONTHLY
  if(cfg.monthly&&String(now.getDate())===cfg.monthlyDate&&nowHM===cfg.monthlyTime){
    allUsers.forEach(u=>{
      const key='monthly_'+u.username+'_'+monthKey;
      if(log[key])return;
      const myMonthly=freshTasks.filter(t=>t.username===u.username&&t.taskType==='monthly'&&t.dueDate&&t.dueDate.startsWith(monthKey));
      if(myMonthly.length===0)return;
      sendTaskEmail(cfg,u,"This Month's Monthly Tasks",myMonthly).then(()=>{},()=>{});
      log[key]=true;logChanged=true;
    });
  }
  if(logChanged)saveEmailLog(log);
}

function getWeekKey(d){
  // ISO-ish week key: year + week number, so each calendar week gets one email max
  const date=new Date(d.getTime());
  date.setHours(0,0,0,0);
  date.setDate(date.getDate()-date.getDay()); // back to Sunday
  const yearStart=new Date(date.getFullYear(),0,1);
  const weekNum=Math.ceil((((date-yearStart)/86400000)+yearStart.getDay()+1)/7);
  return date.getFullYear()+'-W'+weekNum;
}

function startEmailScheduler(){
  runEmailScheduler(); // check once immediately in case the minute already matches
  setInterval(runEmailScheduler,60000);
}


function startReminder(){
  setInterval(()=>{
    if(!CU)return;
    const myTasks=tasks.filter(t=>t.username===CU.username&&t.reminder&&!t.done&&t.dueDate&&t.dueTime);
    const now=new Date();
    myTasks.forEach(t=>{
      if(t.status==='Done')return;
      const due=new Date(t.dueDate+'T'+t.dueTime);
      const diff=(due-now)/60000;
      if(diff>0&&diff<=30){
        if(Notification.permission==='granted'){
          new Notification('ISS Task Pro',{body:`"${t.title}" due at ${t.dueTime}`,icon:'📋'});
        }else{
          toast(`⏰ Reminder: "${t.title}" due at ${t.dueTime}`);
        }
      }
    });
  },60000);
  if(Notification.permission!=='granted'&&Notification.permission!=='denied'){
    Notification.requestPermission();
  }
}
